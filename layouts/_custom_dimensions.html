Custom dimensions

Meta tags for Google Analytics custom dimensions to be included in the tag.

Jinja

{%
  with
  custom_dimensions = [
    {
        "data_value": "save_and_refine_search", 
        "data_id": 8
    }
],
  request = {
    "cookies": ""
},
  current_user = {
    "supplier_organisation_size": "small", 
    "user_research_opted_in": false, 
    "role": "buyer"
}
%}
  {% include "toolkit/layouts/_custom_dimensions.html" %}
{% endwith %}

HTML

  <meta name="ga_customDimension" data-id="10" data-value="buyer">




  <meta name="ga_customDimension" data-id="11" data-value="small">




  <meta name="ga_customDimension" data-id="12" data-value="User research banner">





  <meta name="ga_customDimension" data-id="8" data-value="save_and_refine_search">
